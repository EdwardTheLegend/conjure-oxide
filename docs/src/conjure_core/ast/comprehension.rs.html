53    pub fn domain_of(&self) -> Option<Domain> {
54        let return_expr_domain = self
55            .return_expression_submodel
56            .clone()
57            .into_single_expression()
58            .domain_of()?;
59
60        // return a list (matrix with index domain int(1..)) of return_expr elements
61        Some(Domain::Matrix(
62            Box::new(return_expr_domain),
63            vec![Domain::Int(vec![Range::UnboundedR(1)])],
64        ))
65    }

67    /// Solves this comprehension using Minion, returning the resulting expressions.
68    ///
69    /// If successful, this modifies the symbol table given to add aux-variables needed inside the
70    /// expanded expressions.
71    pub fn solve_with_minion(
72        self,
73        symtab: &mut SymbolTable,
74    ) -> Result<Vec<Expression>, SolverError> {
75        let minion = Solver::new(crate::solver::adaptors::Minion::new());
76        // FIXME: weave proper context through
77        let mut model = Model::new(Arc::new(RwLock::new(Context::default())));
78
79        // only branch on the induction variables.
80        model.search_order = Some(self.induction_vars.clone());
81
82        *model.as_submodel_mut() = self.generator_submodel.clone();
83
84        let minion = minion.load_model(model.clone())?;
85
86        let values = Arc::new(Mutex::new(Vec::new()));
87        let values_ptr = Arc::clone(&values);
88
89        tracing::debug!(model=%model.clone(),comprehension=%self.clone(),"Minion solving comprehension");
90        minion.solve(Box::new(move |sols| {
91            // TODO: deal with represented names if induction variables are abslits.
92            let values = &mut *values_ptr.lock().unwrap();
93            values.push(sols);
94            true
95        }))?;
96
97        let values = values.lock().unwrap().clone();
98
99        let mut return_expressions = vec![];
100
101        for value in values {
102            // convert back to an expression
103
104            let return_expression_submodel = self.return_expression_submodel.clone();
105            let child_symtab = return_expression_submodel.symbols().clone();
106            let return_expression = return_expression_submodel.into_single_expression();
107
108            // we only want to substitute induction variables.
109            // (definitely not machine names, as they mean something different in this scope!)
110            let value: HashMap<_, _> = value
111                .into_iter()
112                .filter(|(n, _)| self.induction_vars.contains(n))
113                .collect();
114
115            let value_ptr = Arc::new(value);
116            let value_ptr_2 = Arc::clone(&value_ptr);
117
118            // substitute in the values for the induction variables
119            let return_expression = return_expression.transform_bi(Arc::new(move |x: Atom| {
120                let Atom::Reference(ref ptr) = x else {
121                    return x;
122                };
123
124                // is this referencing an induction var?
125                let Some(lit) = value_ptr_2.get(&ptr.name()) else {
126                    return x;
127                };
128
129                Atom::Literal(lit.clone())
130            }));
131
132            // Copy the return expression's symbols into parent scope.
133
134            // For variables in the return expression with machine names, create new declarations
135            // for them in the parent symbol table, so that the machine names used are unique.
136            //
137            // Store the declaration translations in `machine_name_translations`.
138            // These are stored as a map of (old declaration id) -> (new declaration ptr), as
139            // declaration pointers do not implement hash.
140            //
141            let mut machine_name_translations: HashMap<ObjId, DeclarationPtr> = HashMap::new();
142
143            // Populate `machine_name_translations`
144            for (name, decl) in child_symtab.into_iter_local() {
145                // do not add givens for induction vars to the parent symbol table.
146                if value_ptr.get(&name).is_some()
147                    && matches!(&decl.kind() as &DeclarationKind, DeclarationKind::Given(_))
148                {
149                    continue;
150                }
151
152                let Name::Machine(_) = &name else {
153                    bug!(
154                        "the symbol table of the return expression of a comprehension should only contain machine names"
155                    );
156                };
157
158                let id = decl.id();
159                let new_decl = symtab.gensym(&decl.domain().unwrap());
160
161                machine_name_translations.insert(id, new_decl);
162            }
163
164            // Update references to use the new delcarations.
165            #[allow(clippy::arc_with_non_send_sync)]
166            let return_expression = return_expression.transform_bi(Arc::new(move |atom: Atom| {
167                if let Atom::Reference(ref decl) = atom
168                    && let id = decl.id()
169                    && let Some(new_decl) = machine_name_translations.get(&id)
170                {
171                    Atom::Reference(new_decl.clone())
172                } else {
173                    atom
174                }
175            }));
176
177            return_expressions.push(return_expression);
178        }
179        Ok(return_expressions)
180    }

182    pub fn return_expression(self) -> Expression {
183        self.return_expression_submodel.into_single_expression()
184    }

186    pub fn replace_return_expression(&mut self, new_expr: Expression) {
187        let new_expr = match new_expr {
188            Expression::And(_, exprs) if exprs.clone().unwrap_list().is_some() => {
189                Expression::Root(Metadata::new(), exprs.unwrap_list().unwrap())
190            }
191            expr => Expression::Root(Metadata::new(), vec![expr]),
192        };
193
194        *self.return_expression_submodel.root_mut_unchecked() = new_expr;
195    }

197    /// Adds a guard to the comprehension. Returns false if the guard does not only reference induction variables.
198    pub fn add_induction_guard(&mut self, guard: Expression) -> bool {
199        if self.is_induction_guard(&guard) {
200            self.generator_submodel.add_constraint(guard);
201            true
202        } else {
203            false
204        }
205    }

207    /// True iff expr only references induction variables.
208    pub fn is_induction_guard(&self, expr: &Expression) -> bool {
209        is_induction_guard(&(self.induction_vars.clone().into_iter().collect()), expr)
210    }

213impl Display for Comprehension {
214    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
215        let generators: String = self
216            .generator_submodel
217            .symbols()
218            .clone()
219            .into_iter_local()
220            .map(|(name, decl): (Name, DeclarationPtr)| {
221                let domain: Domain = decl.domain().unwrap().clone();
222                (name, domain)
223            })
224            .map(|(name, domain): (Name, Domain)| format!("{name}: {domain}"))
225            .join(",");
226
227        let guards = self
228            .generator_submodel
229            .constraints()
230            .iter()
231            .map(|x| format!("{x}"))
232            .join(",");
233
234        let generators_and_guards = itertools::join([generators, guards], ",");
235
236        let expression = &self.return_expression_submodel;
237        write!(f, "[{expression} | {generators_and_guards}]")
238    }

282    pub fn generator(mut self, declaration: DeclarationPtr) -> Self {
283        let name = declaration.name().clone();
284        let domain = declaration.domain().unwrap().clone();
285        assert!(!self.induction_variables.contains(&name));
286
287        self.induction_variables.insert(name.clone());
288
289        // insert into generator symbol table as a variable
290        (*self.generator_symboltable)
291            .borrow_mut()
292            .insert(declaration.clone());
293
294        // insert into return expression symbol table as a given
295        (*self.return_expr_symboltable)
296            .borrow_mut()
297            .insert(DeclarationPtr::new_given(name, domain));
298
299        self
300    }

302    /// Creates a comprehension with the given return expression.
303    ///
304    /// If a comprehension kind is not given, comprehension guards containing decision variables
305    /// are invalid, and will cause a panic.
306    pub fn with_return_value(
307        self,
308        mut expression: Expression,
309        comprehension_kind: Option<ComprehensionKind>,
310    ) -> Comprehension {
311        let parent_symboltable = self
312            .generator_symboltable
313            .as_ref()
314            .borrow_mut()
315            .parent_mut_unchecked()
316            .clone()
317            .unwrap();
318        let mut generator_submodel = SubModel::new(parent_symboltable.clone());
319        let mut return_expression_submodel = SubModel::new(parent_symboltable.clone());
320
321        *generator_submodel.symbols_ptr_unchecked_mut() = self.generator_symboltable;
322        *return_expression_submodel.symbols_ptr_unchecked_mut() = self.return_expr_symboltable;
323
324        // TODO:also allow guards that reference lettings and givens.
325
326        let induction_variables = self.induction_variables;
327
328        // only guards referencing induction variables can go inside the comprehension
329        let (mut induction_guards, mut other_guards): (Vec<_>, Vec<_>) = self
330            .guards
331            .into_iter()
332            .partition(|x| is_induction_guard(&induction_variables, x));
333
334        let induction_variables_2 = induction_variables.clone();
335        let generator_symboltable_ptr = generator_submodel.symbols_ptr_unchecked().clone();
336
337        // fix induction guard pointers so that they all point to variables in the generator model
338        induction_guards = Biplate::<DeclarationPtr>::transform_bi(
339            &induction_guards,
340            Arc::new(move |decl| {
341                if induction_variables_2.contains(&decl.name()) {
342                    (*generator_symboltable_ptr)
343                        .borrow()
344                        .lookup_local(&decl.name())
345                        .unwrap()
346                } else {
347                    decl
348                }
349            }),
350        )
351        .into_iter()
352        .collect_vec();
353
354        let induction_variables_2 = induction_variables.clone();
355        let return_expr_symboltable_ptr =
356            return_expression_submodel.symbols_ptr_unchecked().clone();
357
358        // fix other guard pointers so that they all point to variables in the return expr model
359        other_guards = Biplate::<DeclarationPtr>::transform_bi(
360            &other_guards,
361            Arc::new(move |decl| {
362                if induction_variables_2.contains(&decl.name()) {
363                    (*return_expr_symboltable_ptr)
364                        .borrow()
365                        .lookup_local(&decl.name())
366                        .unwrap()
367                } else {
368                    decl
369                }
370            }),
371        )
372        .into_iter()
373        .collect_vec();
374
375        // handle guards that reference non-induction variables
376        if !other_guards.is_empty() {
377            let comprehension_kind = comprehension_kind.expect(
378                "if any guards reference decision variables, a comprehension kind should be given",
379            );
380
381            let guard_expr = match other_guards.as_slice() {
382                [x] => x.clone(),
383                xs => Expression::And(Metadata::new(), Box::new(into_matrix_expr!(xs.to_vec()))),
384            };
385
386            expression = match comprehension_kind {
387                ComprehensionKind::And => {
388                    Expression::Imply(Metadata::new(), Box::new(guard_expr), Box::new(expression))
389                }
390                ComprehensionKind::Or => Expression::And(
391                    Metadata::new(),
392                    Box::new(Expression::And(
393                        Metadata::new(),
394                        Box::new(matrix_expr![guard_expr, expression]),
395                    )),
396                ),
397
398                ComprehensionKind::Sum => {
399                    panic!("guards that reference decision variables not yet implemented for sum");
400                }
401            }
402        }
403
404        generator_submodel.add_constraints(induction_guards);
405
406        return_expression_submodel.add_constraint(expression);
407
408        Comprehension {
409            return_expression_submodel,
410            generator_submodel,
411            induction_vars: induction_variables.into_iter().collect_vec(),
412        }
413    }
414}

416/// True iff the guard only references induction variables.
417fn is_induction_guard(induction_variables: &HashSet<Name>, guard: &Expression) -> bool {
418    guard
419        .universe_bi()
420        .iter()
421        .all(|x| induction_variables.contains(x))
